# Desarrollo: PostgreSQL local + hot-reload backend

services:
  db:
    image: postgres:17-alpine
    environment:
      - POSTGRES_DB=sipra
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=postgres
    ports:
      - "5432:5432"
    volumes:
      - postgres_data_dev:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres -d sipra"]
      interval: 5s
      timeout: 5s
      retries: 10

  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    environment:
      - DJANGO_SETTINGS_MODULE=config.settings
      - DB_PROVIDER=postgres
      - DATABASE_URL=postgresql://postgres:postgres@db:5432/sipra
      - DEBUG=1
      - SECRET_KEY=dev-key
    ports:
      - "8001:8001"
    volumes:
      - ./backend:/app
    depends_on:
      db:
        condition: service_healthy
    command: >
      sh -c "python manage.py migrate --noinput &&
             python manage.py cargar_secretarias 2>/dev/null || true &&
             python manage.py runserver 0.0.0.0:8001"
