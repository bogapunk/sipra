docker network create red_postgres
✅ 2. Eliminar contenedores
docker rm -f postgres
docker rm -f pgadmin
✅ 3. Crear PostgreSQL en esa red
docker run -d \
--name postgres \
--network red_postgres \
-e POSTGRES_USER=admin \
-e POSTGRES_PASSWORD='qlwpuc#EEVfs6zY4' \
-e POSTGRES_DB=sipra \
-p 5433:5432 \
-v postgres_data:/var/lib/postgresql/data \
